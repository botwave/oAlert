特性
  • 简单的配置即可实现强大的提示框功能，自定义功能强大
  • 内容支持异步加载
  • 内置8种外观供选择，包含：Facebook、Twitter、Google、OSX等样式
  • 可通过修改CSS可实现个性化外观
  • 兼容多种浏览器：Chrome、Safari、Firefox、Opera、IE
  • 兼容IE6哦，亲！╮(╯_╰)╭

自定义按钮
oAlert!
new oAlert({
  style: {
    className: ''
  },
  content: {
    title: 'oAlert',
    text: 'Hello World!',
    buttons: [
      // 普通的按钮，点击后关闭提示框，激活回调方法
      { label: '确定', code: 'yes' },
      // 平面化危险操作按钮的样式，降低误点击的可能
      { label: '取消', code: 'no', flat: true }, 
      // 点击按钮不关闭提示框
      { label: '不关闭', code: 'still', hide: false } 
    ]
  },
  callbacks: {
    buttonClick: function(code) {
      $('#custom_button_label').html('点击了按钮:' + code);
    }
  },
  hide: {
    auto: false
  }
});
自定义样式
oAlert!
new oAlert({
  style: {
    className: '',
    opacity: 0.8,    // 提示框透明度
    minWidth: 500,   // 最小宽度
    minHeight: 150   // 最小高度
  },
  content: {
    title: 'oAlert',
    text: 'Hello World!',
    buttons: [
      { label: '确定', code: 'yes' },
      { label: '取消', code: 'no', flat: true }
    ]
  },
  position: {
    top: 0.5          // 位置，0.5为正中间
  },
  show: {
    delay: 500,      // 延迟0.5秒出现
    duration: 1000    // 过渡动画时长为1秒
  },
  hide: {
    delay: 10000,     // 10秒后自动关闭
    duration: 700     // 过渡动画时长为0.7秒
  },
  overlay: {
    style: {
      color: '#fff',  // 覆盖层颜色为白色
      opacity: .2     // 覆盖层透明度为0.2
    },
    show: {
      duration: 500,  // 覆盖层过渡动画时长为0.5秒
    },
    hide: {
      duration: 1400  // 覆盖层过渡动画时长为1.4秒
    }
  }
});
回调方法
oAlert!
new oAlert({
  style: {
    className: ''
  },
  content: {
    title: 'oAlert',
    text: 'Hello World!',
    buttons: [
      { label: '关闭' }
    ]
  },
  callbacks: {
    beforeRender: function() {
      $('#callbacks_label').html('开始渲染');
    },
    render: function() {
      $('#callbacks_label').append('渲染完成');
    },
    beforeReposition: function() {
      $('#callbacks_label').append('重定位开始');
    },
    reposition: function() {
      $('#callbacks_label').append('重定位完成');
    },
    beforeShow: function() {
      $('#callbacks_label').append('开始显示');
    },
    show: function() {
      $('#callbacks_label').append('显示完成');
    },
    beforeHide: function() {
      $('#callbacks_label').append('开始关闭');
    },
    hide: function() {
      $('#callbacks_label').append('关闭完成');

异步加载内容
oAlert!
new oAlert({
  style: {
    className: ''
  },
  content: {
    ajax: {
      url: 'res/user.json', // 请求地址
      type: 'GET', // 请求方法，'GET'或'POST'
      data: { uid: 1 }, // 参数值
      dataType: 'json', // 数据格式，默认为json
      delay: 2000,  // 延迟2秒发送请求
      loading: true, // true为显示加载过程，false为加载完毕才显示
      success: function(data, status) { // 请求成功回调方法
        var user = data.user; // data为返回的数据
        this.set('content', {
          title: '用户资料',
          text: '编号：' + user.uid + '\n' +
                '用户名：' + user.name + '\n' +
                '邮箱：' + user.email + '\n' +
                '<image src="' + user.avatar + '"/>',
          buttons: [
            { label: '关闭' }
          ]
        });
      },
      error: function(status, errorThrown) { // 请求失败回调方法
        this.set('content.title', '加载失败');
        this.set('content.text', '加载内容失败，请重试');
        this.set('content.buttons', [{ label: '关闭' }]);
      }
    }
  },
  hide: {
    delay: 120000
  }
});
文本输入框
oAlert!
new oAlert({
  style: {
    className: ''
  },
  content: {
    title: '用户资料',
    text: '请输入您的相关资料',
    buttons: [
      { label: '确定', code: 'yes' },
      { label: '取消', code: 'no', flat: true }
    ],
    inputs: [
      // 不被创建的输入框
      { field: 'wont-be-created', create: false, 
        defaultValue: '不会创建我' },
      // 带默认值的输入框
      { field: 'name', label: '用户名', 
        defaultValue: 'nino zhang' },
      // 密码输入框
      { field: 'password', label: '密码：', 
        type: 'password' },
      // 只读输入框
      { field: 'age', label: '年龄（只读）', 
        defaultValue: '18岁', readOnly: true },
      // 限制最多可输入字符数的输入框
      { field: 'tag', label: '我的标签（最长10个字符）', 
        maxLength: 10 },
      // 失效的输入框
      { field: 'disabled', label: '不给你输入：', 
        defaultValue: '娃哈哈哈哈', disabled: true }
    ]
  },
  callbacks: {
    buttonClick: function(code, values) {
      var label = $('#custom_input_label').empty();
      label.append('点击了按钮:' + code + '<br/>');
      // 输入框的值保存在values中
      for(var key in values) {
        label.append(key + ': ' + values[key] + '<br/>');
      }
    }
  },
  hide: {
    auto: false
  }
});
